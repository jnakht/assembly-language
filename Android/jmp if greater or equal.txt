INCLUDE 'EMU8086.INC'
.MODEL SMALL
.STACK 100H
.DATA
.CODE
MAIN PROC           
    PRINT 'ENTER THE FIRST NUMBER='
    MOV AH,01H
    INT 21H
    MOV BL,AL
    MOV DL,10
    MOV AH,02H
    INT 21H
    MOV DL,13
    MOV AH,02H
    INT 21H
    
     PRINT 'ENTER THE SECOND NUMBER='
     MOV AH,01H
     INT 21H
     MOV CL,AL
     
    MOV DL,10
    MOV AH,02H
    INT 21H
    MOV DL,13
    MOV AH,02H
    INT 21H
   CMP BL, CL
JGE GE_LABEL
PRINT 'FIRST IS LESS$'
MOV AH,4CH
    INT 21H 
     EXIT:

GE_LABEL:
PRINT 'FIRST IS GREATER OR EQUAL$' 
MOV AH,4CH
    INT 21H

   MAIN ENDP
END MAIN
